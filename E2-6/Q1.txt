問１：

  アンケート回答テーブル(quest)と社員テーブル(employee)から
  女性の氏名(カナ)だけを、
  氏(カナ)、名(カナ)昇順で取り出して見ましょう。
  なお、重複データがあってもそのまま出力するものとします。
  以下の空欄を埋めて、SQL命令を完成させてください。

  SELECT
    q.name_kana
  FROM
    quest AS q
  WHERE
    (①)
  (②)
    SELECT
      (③)
    FROM
      employee AS e
    WHERE
      e.sex = 2
    ORDER BY
      (④)
    ;


回答：

  ①： quest.sex = '女'
  ②： UNION ALL
  ③： CONCAT(employee.l_name_kana,employee.f_name_kana)
  ④： 1
  

結果：

  SELECT                                     -- アンケート回答者氏名(カナ)のマージ結果を見るために以下カラムを設定する
    q.name_kana                              -- 氏名(カナ)カラムを設定する
  FROM                                       -- ベースとなるテーブルを指定する
    quest AS q                               -- アンケート回答テーブルを指定する
  WHERE                                      -- 性別が女性であるデータで絞り込むために以下条件を指定する
    q.sex = '女'                             -- sexフィールドが '女' のデータで絞り込む
  UNION ALL                                  -- 重複分も表示するように次の結果をマージます
    SELECT                                   -- 社員氏名(カナ)を見るために以下のカラムを指定します
      CONCAT(e.l_name_kana,e.f_name_kana)    -- 社員氏名(カナ)カラムを設定する
    FROM                                     -- 社員氏名を取得するために以下テーブルを指定する
      employee AS e                          -- 社員テーブルを指定する
    WHERE                                    -- 女性社員の情報をみるため以下条件で絞り込みをかける
      e.sex = 2                              -- sexフィールドが 2(女性) のデータで絞り込む
    ORDER BY                                 -- 統合する列を指定する
      1 ASC                                  -- 1列のみなので 1 を昇順で指定する
  ;
