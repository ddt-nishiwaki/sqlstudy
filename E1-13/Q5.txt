問５：

  以下は、社員テーブル(employee)から社員名(氏+名を連結したもの)、
  役職クラス(部長、課長は管理職、主任、担当は総合職、アシスタントは一般職とします)を出力するためのSQL命令ですが、
  2つ誤りがあります。
  誤りを指摘してください。

  SELECT
    CONCAT(l_name, f_name),
    CASE
      WHERE class IN ('部長','課長') THEN '管理職'
      WHERE class IN ('主任','担当') THEN '総合職'
      ELSE THEN '一般職'
    END AS 役職クラス
  FROM
    employee
  ;


回答：

  SELECT
    CONCAT(l_name, f_name),
    CASE
      WHERE class IN ('部長','課長') THEN '管理職'   <---- WHEN をWHERE と間違えている
      WHERE class IN ('主任','担当') THEN '総合職'   <---- WHEN をWHERE と間違えている
      ELSE THEN '一般職'                             <---- ELSE の後に THEN はいらない
    END AS 役職クラス                                 
  FROM
    employee
  ;


結果：

  SELECT                                             -- 各社員の役職クラスを表示するために以下のカラムを出力する
    CONCAT(l_name, f_name),                          -- 社員の氏名を出力するカラム
    CASE                                             -- 役職クラスを表示するカラム
      WHEN class IN ('部長','課長') THEN '管理職'    -- 部長、課長は管理職として表示する
      WHEN class IN ('主任','担当') THEN '総合職'    -- 主任、担当は総合職として表示する
      ELSE '一般職'                                  -- それ以外は一般職として表示する
    END AS 役職クラス                                -- 役職クラスとして出力する
  FROM                                               -- 各社員の役職クラスを表示するために以下のテーブルから抽出する
    employee                                         -- 社員テーブル(employee)を指定する
  ;
